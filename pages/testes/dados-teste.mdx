function validarTelefone(telefone) {
  // Remove caracteres n√£o num√©ricos
  const numero = telefone.replace(/\D/g, '');

  // Padr√µes v√°lidos:
  // +5511999999999 (13 d√≠gitos com c√≥digo pa√≠s)
  // 11999999999 (11 d√≠gitos sem c√≥digo pa√≠s)
  // 1199999-9999 (formatado)

  const patterns = [
    /^55\d{2}9\d{8}$/, // +55 + DDD + 9 + 8 d√≠gitos
    /^\d{2}9\d{8}$/,   // DDD + 9 + 8 d√≠gitos

    /^\d{2}\d{8}$/     // DDD + 8 d√≠gitos (fixo)

// Formatador de telefone
function formatarTelefone(telefone) {
  const numero = telefone.replace(/\D/g, '');

  if (numero.length === 13 && numero.startsWith('55')) {
    // +55 11 99999-9999

    // (11) 99999-9999
    return `(${numero.slice(0, 2)}) ${numero.slice(2, 7)}-${numero.slice(7)}`;
  }

class CPFGenerator {
  static generate() {
    // Gera 9 primeiros d√≠gitos aleat√≥rios
    const digits = Array.from({ length: 9 }, () =>
      Math.floor(Math.random() * 10)
    );

    // Calcula primeiro d√≠gito verificador
    let sum = 0;
    for (let i = 0; i < 9; i++) {
      sum += digits[i] * (10 - i);
    }
    const firstDigit = (sum * 10) % 11;
    digits.push(firstDigit >= 10 ? 0 : firstDigit);

    // Calcula segundo d√≠gito verificador
    sum = 0;
    for (let i = 0; i < 10; i++) {
      sum += digits[i] * (11 - i);
    }
    const secondDigit = (sum * 10) % 11;
    digits.push(secondDigit >= 10 ? 0 : secondDigit);

    return digits.join('');
  }

  static format(cpf) {
    return cpf.replace(/(\d{3})(\d{3})(\d{3})(\d{2})/, '$1.$2.$3-$4');
  }

  static generateFormatted() {
    return this.format(this.generate());
  }
}

class CNPJGenerator {
  static generate() {
    // Gera 12 primeiros d√≠gitos (8 + /0001)
    const base = Array.from({ length: 8 }, () =>
      Math.floor(Math.random() * 10)
    );
    base.push(0, 0, 0, 1); // Adiciona /0001

    // Calcula primeiro d√≠gito verificador
    const weights1 = [5, 4, 3, 2, 9, 8, 7, 6, 5, 4, 3, 2];
    let sum = 0;
    for (let i = 0; i < 12; i++) {
      sum += base[i] * weights1[i];
    }
    const firstDigit = sum % 11;
    base.push(firstDigit < 2 ? 0 : 11 - firstDigit);

    // Calcula segundo d√≠gito verificador
    const weights2 = [6, 5, 4, 3, 2, 9, 8, 7, 6, 5, 4, 3, 2];
    sum = 0;
    for (let i = 0; i < 13; i++) {
      sum += base[i] * weights2[i];
    }
    const secondDigit = sum % 11;
    base.push(secondDigit < 2 ? 0 : 11 - secondDigit);

    return base.join('');
  }

  static format(cnpj) {
    return cnpj.replace(/(\d{2})(\d{3})(\d{3})(\d{4})(\d{2})/, '$1.$2.$3/$4-$5');
  }

  static generateFormatted() {
    return this.format(this.generate());
